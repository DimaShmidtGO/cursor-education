Base

git config --global user.name DimaShmidtGO - встановлює ім'я користувача

git config --global user.email dimashmidtm@gmail.com - встановлює Email користувача

git init - Створює локальний репрозиторій

git status - Показує всі файли та їхній статус

git add - Додати один файл (Навзва файлу) або додає всі файли (.)

git commit - Зберігає зміни в файлі та робить коментар з описом, так званих комітів

git log - Журнал комітів

git checkout - переключається між гілками і перевіряє данні

git branch - показує гілки, де можна додати, видалити і змінити назву

git push - збирає всі зміни для вигрузки на відаллений репрозиторій

git pull - завантажує всі зміни з відалленого репрозиторію

git merge - Об'єднює одну або більше гілок в одну

git clone - клонує відаллений репрозиторій

.gitignore - вказуємо GIT, які файли потрібно ігнорувати

Advanced

git revert - скасовує існуючий коміт

git reset - видаляє до останнього коміту

git rebase - переносить зміни з однієї гілки в іншу

git cherry-pick - переносить комміти з одного місця інше

git stash - відкладає зміни в архів
